@name RailDriver/lib/smartEntityManagement

function void semLockE2()
{
    local E = entity()
    local LockE2 = 0

    if (isSinglePlayer() != 0)
    {
        LockE2 = 0
    }

    else
    {
        LockE2 = 1
    }

    E:propNotSolid(LockE2)
    E:propDraw(!LockE2)
    E:propShadow(!LockE2)
}

function number semSmartParentE2toLocoBody()
{
    local E = entity()

    if (!E:parent())
    {
        local W = E:isWeldedTo()
        if (W:isValid() == 0 | !W)
        {
            return 0
        }

        E:constraintBreak("weld", W)
        E:parentTo(W)
    }

    return 1
}
